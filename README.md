# Perhitungan-Gaji

    #include<iostream>
    using namespace std;
    int main()
    {
    int upah,gaji;
    cout << "\t\t ================================= \n";
    cout << "\t\t Program menghitung gaji karyawan \n";
    cout << "\t\t ================================= \n";
    cout << "\t\t Berapa jam karyawan bekerja : ";

    gaji=upah*5000;
    cout << "\t\t Bayaran untuk karyawan adalah sebesar = Rp " << gaji << endl;
    return 0;
    }
